数据结构：排序算法

Posted on 2021-10-24

排序算法：冒泡排序，简单选择排序，直接插入排序，希尔排序，堆排序，归并排序，快速排序。

1.冒泡排序

冒泡排序时间复杂度O(n^2);

基础冒泡排序
void BubbleSort0(SqList *L)
{
    for(int i=1;i<L->length;i++)
    {
        for(int j=i+1;j<=L->lenght;j++)
        {
            if(L->r[i]>L->r[j])
            {
                swap(L,i,j);
            }
        }
    }
}
正宗冒泡排序
void BubbleSort(SqList *L)
{
    bool flag=true;
    for(int i=1;i<L->length;i++)
    {    
        flag=false;
        for(int j=L->length&&flag;j>i;j--)
        {    
            if(L->r[j]<L->r[j-1])
            {
                flag=true;
                swap(L,j,j-1);
            }
        }
    }
}

2.简单选择排序

简单排序时间复杂度为O(n^2),但是简单排序的性能略优于冒泡排序。

void SelectSort(SqList *L)
{
    for(int i=1;i<L->length;i++)
    {
        int min=i;
        for(int j=i+1;j<=L->length;j++)
        {
            if(L->r[min]>L->r[j])
            {
                min=j;
            }
        }
        if(min!=i)
        {
            swap(L,i,min);
        }
    }
}

3.直接插入排序

直接插入排序时间复杂度为O(n^2),性能优于前两者。

void InsertSort(SqList *L)
{
    for(int i=2;i<=L->length;i++)
    {
        if(L->r[i]<L->r[i-1])
        {
            L->r[0]=L->r[j];
            for(int j=i-1;L->r[j]>L->r[0];j--)
            {
                L->r[j+1]=L->r[j];
            }
            L->r[j+1]=L->r[0];
        }
    }
}

4.希尔排序

插入排序的进阶算法 时间复杂度为O(n^(3/2))。

void ShellSort(SqList *L)
{
    int increment;
    do
    {
        increment=increment/3+1;
        for(int i=increment+1;i<=L->length;i++)
        {
            if(L->r[i]<L->r[i-incrment])
            {
                L->r[0]=L->r[i];
                for(int j=i-increment;j>0&&L->r[0]<L->r[j];j-=increment)
                {
                    L->r[j+increment]=r[j];
                }
                L->r[j+increment]=L->r[0];
            }
        }
    }while(increment>1)
}

5.堆排序

堆是具有下列性质的完全二叉树：每个结点的值都大于或等于其左右孩子结点的值，称为大顶堆，每个结点的值都小于或等于其左右孩子结点的值，称为小顶堆。

void HeapSort(SqList *L)
{
    for(int i=L->length/2;i>0;i--)
    {
        HeapAdjust(L,i,L->length);
    }
    
    for(int i=L->lenght;i>1;i--)
    {
        swap(L,1,i);
        HeapAdjust(L,1,i-1);
    }
}

void HeapSort(SqList *L,int s,int m)
{
    int temp,j;
    temp=L->r[s];
    for(j=2*s;j<m;j*=2)
    {
        if(j<m&&L->r[j]<L->r[j+1])
            ++j;
        if(temp>=L->r[j])
            break;
        L->r[s]=L->r[j]
        s=j;
    }
    L->r[s]=temp;
}
